panelTaskAndAlertController.js
// 'use strict';
/**
 * @ngdoc function
 * @name PtEMobile.controller:PanelTaskAndAlertCtrl
 * @description
 * # PanelTaskAndAlertCtrl
 * Controller of the PtEMobile
 * This controller has added from PTE-6607
 */
angular.module("PtEMobile").controller("PanelTaskAndAlertCtrl", function (
  $scope,
  $rootScope,
  $location,
  $sce,
  $translate,
  $window,
  api,
  sessionFactory,
  $q,
  commonService,
  taskFactory
) {
  console.log('=========== PanelTaskAndAlertCtrl ===========');

  $scope.isLoadingTasksAndAlerts = true;
  $scope.filter = { of: 'only' }
  $scope.tasksAndAlerts = [];
  $scope.totalTasks = 0;
  $scope.skip = 0;
  $scope.pageSize = 10;
  $scope.pageSelected = 1;
  $scope.sorting = {
    sortType: 'CreatedDate',
    sortReverse: true
  };
  $scope.showLoadMore = false;
  $scope.isMyOpenTasksAndAlerts = true;

  $scope.fetchTasksAndAlerts = _.debounce(function(patientDataView) {
    fetchTasksAndAlerts(patientDataView);
  }, 300);

  $scope.showPopupAddEditTask = function (task, mode) {
    console.log('showPopupAddEditTask...')
    const patientDataView = _.cloneDeep($scope.patientDataView);//7429
    $rootScope.showPopupAddEditTask(task, mode, result => {
      result && getTasksAndAlerts();
    }, patientDataView);//7429
  }

  $scope.showPopupDeleteTask = function (task) {
    console.log('showPopupDeleteTask...', task);
    commonService.popupMessage(
      $translate.instant('Common.Confirmation'),
      $translate.instant('Task.DeleteTaskConfirmation'),
      // handle cancel
      () => { },
      // handle confirm
      () => {
        taskFactory.deleteTask(task._id).then(result => {
          getTasksAndAlerts();
          commonService.showToast(
            $translate.instant("Task.DeleteTaskSuccessfully"),
            5000,
            null,
            null,
            false,
            null,
            null,
            null
          );
        }, err => {
          console.log('Cannot delete task', err);
          getTasksAndAlerts();
        })
      },
      {
        cancelLabel: $translate.instant('Common.Cancel'),
        confirmLabel: $translate.instant('Common.Yes'),
        popupSize: 'small',
        isNoHeaderBackgoundColor: false,
        isHeaderCenter: $rootScope.screenType == ScreenTypeConst.Phone,
        isActionsFullWidth: $rootScope.screenType == ScreenTypeConst.Phone,
        closeButton: false,
      }
    );
  }

  $scope.updateTaskStatus = function (task, status) {
    console.log('updateTaskStatus...')
    taskFactory.updateTaskStatus(task._id, status).then(result => {
      getTasksAndAlerts();
      commonService.showToast(
        status == 'Resolved' ? $translate.instant("Task.TaskResolvedSuccessfully") : $translate.instant("Task.TaskReopenedSuccessfully"),
        5000,
        null,
        null,
        false,
        null,
        null,
        null
      );
    }, err => {
      console.log('updateTaskStatus error', err);
    })
  }

  $scope.showDropDownAction = function ($event, filterOf) {
    $scope.isLoadingTasksAndAlerts = true;
    $scope.totalTasks = 0;
    $scope.tasksAndAlerts = [];
    $scope.pageSelected = 1;
    $scope.filter.of = filterOf;
    getTasksAndAlerts();
  }

  $scope.loadMore = function () {
    if ($scope.pageSelected >= $scope.totalTasks / $scope.pageSize)
      return;

    $scope.pageSelected += 1;
    getTasksAndAlerts();
  };

  var createTaskTimer;
  $scope.createTask = function () {
    if (!$scope.parentScope || !$scope.parentScope.patientDataView) {
      clearTimeout(createTaskTimer);
      createTaskTimer = setTimeout($scope.createTask, 300);
      return;
    };

    const patientDataView = $scope.parentScope.patientDataView;
    const task = {
      Patient: patientDataView._id,
    }
    $rootScope.showPopupAddEditTask(task, 'create', result => {
      if (result) {
        $scope.pageSelected = 1;
        $scope.totalTasks = 0;
        getTasksAndAlerts()
      }
    })
  }

  var retryAttempt = 0;
  function getTasksAndAlerts() {
    console.log('getTasksAndAlerts...', $scope, 'retryAttempt', retryAttempt)
    const patientDataView = $scope.parentScope && ($scope.parentScope.patientDataView || $scope.parentScope.profile);
    if (!patientDataView) {
      retryAttempt < 10 && setTimeout(() => {
        retryAttempt++;
        getTasksAndAlerts();
      }, 300);
      return;
    }
    console.log('patientDataView...', patientDataView, $scope.filter.of)
    $scope.fetchTasksAndAlerts(patientDataView);
  }

  function fetchTasksAndAlerts(patientDataView) {
    retryAttempt = 0;
    const filterData = {
      myTasks: $scope.filter.of === 'only',
      Statuses: ['Open'],
      Patients: [patientDataView._id]
    }
    if ($scope.pageSelected === 1){
      $scope.totalTasks = 0;
      $scope.tasksAndAlerts = [];
    }
    $scope.isLoadingTasksAndAlerts = true;
    const paging = { page: $scope.pageSelected, pageSize: $scope.pageSize };

    const requestData = {
      filterData: filterData,
      paging: paging,
      sorting: $scope.sorting,
      hasChanged: false,
    }

    const disabledSpinner = true;
    taskFactory.getTasks(requestData, disabledSpinner).then(response => {
      console.log('panelTaskAndAlertController loadTasks... response', response);
      console.log('fetchTasksAndAlerts_$scope.filter.of',$scope.filter.of);
      
      const tasks = taskFactory.parseTasks(response.docs, (task, type) => {
        if (type == 'IncompleteEligibility')
          $scope.tasksAndAlerts = _.filter($scope.tasks, item => item._id !== task._id);
      });
      $scope.tasksAndAlerts = $scope.pageSelected == 1 ? tasks : _.concat($scope.tasksAndAlerts, tasks);
      $scope.totalTasks = response.paging.total;
      $scope.showLoadMore = $scope.totalTasks > $scope.pageSize * $scope.pageSelected;
      expandCollapseSection();
    }).finally(() => {
      $scope.isLoadingTasksAndAlerts = false;
    });
  }

  function expandCollapseSection() {
    const $payAttention = $("#pay-attention")
    if ($scope.tasksAndAlerts && $scope.tasksAndAlerts.length) {
      $('[data-target="#pay-attention"]').attr("aria-expanded", "true");
      $payAttention.hasClass("in") ||
        $payAttention
          .addClass("in")
          .css("height", "");
    } else {
      $('[data-target="#pay-attention"]').attr("aria-expanded", "false");
      $payAttention.removeClass("in");
    }
  }

  function initializer() {
    let dateFormat = ($rootScope.dateFormat || '').replace('DD', 'dd').replace('YYYY', 'yyyy');
    $scope.dateFormat = dateFormat;
    if ($rootScope.timeFormat == 'time12') {
      $scope.dateTimeFormat = dateFormat + ' ' + 'hh:mm a';
    } else {
      $scope.dateTimeFormat = dateFormat + ' ' + 'HH:mm';
    }
    getTasksAndAlerts();

    const refreshTasksAndAlerts = $rootScope.$on('refreshTasksAndAlerts', function (event, args) {
      console.log('refreshTasksAndAlerts...')
      $scope.isLoadingTasksAndAlerts = true;
      $scope.totalTasks = 0;
      $scope.tasksAndAlerts = [];
      $scope.pageSelected = 1;
      $scope.filter.of = 'only';
      getTasksAndAlerts();
    });

    $scope.$on('$destroy', function () {
      console.log('destroy PanelTaskAndAlertCtrl...')
      refreshTasksAndAlerts();
    });
  }

  initializer();
});
